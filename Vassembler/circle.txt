draw_circle:

    call get_y   # получаем y координату

    push rax     #
    push rbx     #
    push 20      # матричная формула
    mult         #
    add          #

    pop rax 
    push 1      
    pop [rax]    

    call center_coord

    push 44
    push 7
    div
    push 40

ret


get_x:
    push rax
    push rcx
    push rdx
    cos          #
    mult         # x = x_0 + cosa
    add          #

    pop rax
ret


get_y:
    push rbx
    push rcx
    push rdx
    sin          #
    mult         # y = y_0 + sina
    floor        #
    add    

    pop rbx
ret


initial_conditions:
    push 6       # радиус
    pop rcx

    push 0       # начальный угол
    pop rdx
ret


center_coord:

    push 10      # y центра
